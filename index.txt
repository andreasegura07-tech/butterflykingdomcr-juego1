<script>

const butterflies=[

{
name:"Morpho Azul",
img:"https://upload.wikimedia.org/wikipedia/commons/1/10/Morpho_menelaus.jpg"
},

{
name:"Monarca",
img:"https://upload.wikimedia.org/wikipedia/commons/4/4c/Monarch_Butterfly.jpg"
},

{
name:"Mariposa Cristal",
img:"https://upload.wikimedia.org/wikipedia/commons/0/0b/Greta_oto.jpg"
}

]

let score=0

function startGame(){

document.getElementById("start").style.display="none"
document.getElementById("game").style.display="flex"

next()

}

function next(){

const b=butterflies[Math.floor(Math.random()*butterflies.length)]

document.getElementById("butterfly").src=b.img

const options=document.getElementById("options")

options.innerHTML=""

butterflies.forEach(o=>{

const btn=document.createElement("button")

btn.className="opt"

btn.innerText=o.name

btn.onclick=()=>{

if(o.name===b.name){

score+=10

}else{

score-=5

}

document.getElementById("score").innerText=score

next()

}

options.appendChild(btn)

})

}

/* registrar service worker */

if('serviceWorker' in navigator){

navigator.serviceWorker.register('service-worker.js')

}

</script>
